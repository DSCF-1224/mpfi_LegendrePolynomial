183static int mpfi_LegendrePolynomial_RecursiveSingleStep(
184 /***/ /****/ mpfi_ptr result , //
185 const /****/ unsigned long degree , //
186 /***/ /****/ mpfi_srcptr range , //
187 /***/ /****/ mpfi_srcptr x , //
188 /***/ /****/ mpfi_srcptr ref1 , //
189 /***/ /****/ mpfi_srcptr ref2 , //
190 /***/ struct mpfi_LegendrePolynomialWorkspace_t *const workspace ) {
191
192 // temp1 <- n
193 mpfi_set_ui( workspace->temp1 , degree );
194
195 // temp2 <- n - 1
196 mpfi_set_ui( workspace->temp2 , (degree - 1UL) );
197
198 // temp3 <- 2n - 1
199 mpfi_set_ui( workspace->temp3 , (degree + degree - 1UL) );
200
201 // temp4 <- ( 2n - 1 ) * x
202 mpfi_mul( workspace->temp4 , workspace->temp3 , x );
203
204 // temp3 <- ( 2n - 1 ) * lp[ n - 1 ] * x
205 mpfi_mul( workspace->temp3 , workspace->temp4 , ref1 );
206
207 // temp3 <- ( n - 1 ) * lp[ n - 2 ]
208 mpfi_mul( workspace->temp4 , workspace->temp2 , ref2 );
209
210 // temp2 <- temp3 - temp4
211 mpfi_sub( workspace->temp2 , workspace->temp3 , workspace->temp4 );
212
213 // result <- ( temp4 - temp3 ) / n
214 mpfi_div( result , workspace->temp2 , workspace->temp1 );
215
216
217 if ( mpfi_nan_p(result) || mpfi_inf_p(result) || !mpfi_is_inside(result, range) ) {
218
219 return EXIT_FAILURE;
220
221 }
222
223 return EXIT_SUCCESS;
224
225}

239 /***/ /****/ mpfi_ptr result , //
240 /***/ /****/ mpfi_ptr ref1 , //
241 /***/ /****/ mpfi_ptr ref2 , //
242 const /****/ unsigned long degree , //

244 const /****/ mpfi_srcptr x , //

246
247

273 if ( i == degree ) break;

275 mpfi_set( ref2, ref1 );
276 mpfi_set( ref1, result );
