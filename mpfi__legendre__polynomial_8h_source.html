165 /***/ /****/ mpfi_ptr result , //
166 const /****/ unsigned long degree , //
167 /***/ /****/ mpfi_srcptr range , //
168 /***/ /****/ mpfi_srcptr x , //
169 /***/ /****/ mpfi_srcptr ref1 , //
170 /***/ /****/ mpfi_srcptr ref2 , //
171 /***/ struct mpfi_LegendrePolynomialWorkspace_t *const workspace ) {
172
173 // https://gitlab.inria.fr/mpfi/mpfi/-/blob/master/src/div.c
174 // https://gitlab.inria.fr/mpfi/mpfi/-/blob/master/src/is_inside.c
175 // https://gitlab.inria.fr/mpfi/mpfi/-/blob/master/src/mul.c
176 // https://gitlab.inria.fr/mpfi/mpfi/-/blob/master/src/predicates.c
177 // https://gitlab.inria.fr/mpfi/mpfi/-/blob/master/src/set_ui.c
178 // https://gitlab.inria.fr/mpfi/mpfi/-/blob/master/src/sub.c
179
180 // temp1 <- n
181 mpfi_set_ui( workspace->temp1 , degree );
182
183 // temp2 <- n - 1
184 mpfi_set_ui( workspace->temp2 , (degree - 1UL) );
185
186 // temp3 <- 2n - 1
187 mpfi_set_ui( workspace->temp3 , (degree + degree - 1UL) );
188
189 // temp4 <- ( 2n - 1 ) * x
190 mpfi_mul( workspace->temp4 , workspace->temp3 , x );
191
192 // temp3 <- ( 2n - 1 ) * lp[ n - 1 ] * x
193 mpfi_mul( workspace->temp3 , workspace->temp4 , ref1 );
194
195 // temp3 <- ( n - 1 ) * lp[ n - 2 ]
196 mpfi_mul( workspace->temp4 , workspace->temp2 , ref2 );
197
198 // temp2 <- temp3 - temp4
199 mpfi_sub( workspace->temp2 , workspace->temp3 , workspace->temp4 );
200
201 // result <- ( temp4 - temp3 ) / n
202 mpfi_div( result , workspace->temp2 , workspace->temp1 );
203
204
205 if ( mpfi_nan_p(result) || mpfi_inf_p(result) || !mpfi_is_inside(result, range) ) {
206
207 return EXIT_FAILURE;
208
209 }
210
211 return EXIT_SUCCESS;
212
213}
